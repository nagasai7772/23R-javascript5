1.SOLVE USING FOR-IN-LOOP
var car={brand:'Toyota',model:'corolla',year:'2020'};
for(names in car)
{
console.log(`${names} : ${car[names].toUpperCase()}`);
 }
O/P: brand:TOYOTA
model:COROLLA
year:2020

2.var numbers=[1,2,3,4,5];
for(position in numbers)
{
console.log(`${numbers[position]}-HI`);
 }
O/P:1-HI
2-HI
3-HI
4-HI
5-HI

3.var fruits=['Apple','Banana','Cherry','Date'];
for(line in fruits)
{
console.log(`${line}-${fruits[line].toUpperCase()}`)
 }
O/P: 0-APPLE
1-BANANA
2-CHERRY
3-DATE
